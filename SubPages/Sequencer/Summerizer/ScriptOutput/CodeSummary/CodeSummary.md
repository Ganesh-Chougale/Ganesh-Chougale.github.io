exportconstdataLoader={
asyncloadCsvData(csvString){
if(!csvString){
console.error("NoCSVstringprovided.");
returnnull;
}
constlines=csvString.trim().split('\n');
if(lines.length<2){
console.error("CSVmustcontainatleastaheaderandonedatarow.");
returnnull;
}
constheaders=lines[0].split(',').map(header=>header.trim());
constdataRows=lines.slice(1).map(line=>{
constvalues=line.split(',').map(value=>value.trim());
constrowObject={};
headers.forEach((header,index)=>{
rowObject[header]=values[index];
});
returnrowObject;
});
return{
headers:headers,
data:dataRows,
originalCsvString:csvString
};
}
};
